class A {
  var x = 1;
  var y = 2;

  function m() {
    this.m2();
  }

  function m2() {
    return x+y;
  }
}

class B extends A {
  var y = 22;
  var z = 3;

  function m() {
    super.m();
  }

  function m2() {
    return x+y+z;
  }
}

class C extends B {
  var y = 222;
  var w = 4;

  function m() {
    super.m();
  }

  static function main() {
    var c = new C();
    c.m();
  }
}
